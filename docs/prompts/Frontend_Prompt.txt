import React, { useState, useRef, useCallback } from 'react';
import { Settings, User, Globe, Hash, Loader2 } from 'lucide-react';
import { createRoot } from 'react-dom/client';

// Import your CSS entry point here so Vite bundles it automatically
// import '../css/app.css';


// --- Configuration ---
const CONFIG = {
    initialRows: 6,
    initialCols: 6,
    items: [
        { id: 'fig-tree', icon: '🌲', title: 'Tree', desc: 'A tall tree, forms a forest when clustered.', count: 10 },
        { id: 'fig-cave', icon: '🕳️', title: 'Cave', desc: 'A dark, mysterious stone cave entrance.', count: 1 },
        { id: 'fig-fox', icon: '🦊', title: 'Fox', desc: 'A clever, orange fox.', count: 1 },
        { id: 'fig-squirrel', icon: '🐿️', title: 'Squirrel', desc: 'A small, energetic squirrel.', count: 1 },
        { id: 'fig-castle', icon: '🏰', title: 'Castle', desc: 'A grand stone castle with towers.', count: 1 },
        { id: 'fig-prince', icon: '🤴', title: 'Prince', desc: 'A young prince looking for adventure.', count: 1 },
        { id: 'fig-princess', icon: '👸', title: 'Princess', desc: 'A wise princess.', count: 1 },
        { id: 'fig-dragon', icon: '🐉', title: 'Dragon', desc: 'A magical dragon.', count: 1 }
    ]
};

const GridWidget = ({ rows = CONFIG.initialRows, cols = CONFIG.initialCols, initialItems = CONFIG.items }) => {
    // State
    const [inventory, setInventory] = useState(initialItems);
    const [board, setBoard] = useState({}); // Map: index -> item
    const [formData, setFormData] = useState({
        age: '2',
        language: 'de',
        narrator: 'Male'
    });
    
    const [isGenerating, setIsGenerating] = useState(false);
    const [generatedOutput, setGeneratedOutput] = useState(null);

    // --- Drag & Drop Handlers ---

    const handleDragStart = (e, item, source, originIndex = null) => {
        const data = {
            id: item.id,
            source: source,
            originIndex: originIndex
        };
        e.dataTransfer.setData('application/json', JSON.stringify(data));
        e.dataTransfer.effectAllowed = 'move';
    };

    const handleDragOver = (e) => {
        e.preventDefault();
        e.dataTransfer.dropEffect = 'move';
    };

    const handleBoardDrop = (e, targetIndex) => {
        e.preventDefault();
        const dataStr = e.dataTransfer.getData('application/json');
        if (!dataStr) return;

        const data = JSON.parse(dataStr);

        // Check if cell is occupied
        if (board[targetIndex]) return;

        let itemToPlace = null;

        if (data.source === 'inventory') {
            const inventoryItem = inventory.find(i => i.id === data.id);
            if (!inventoryItem) return;

            // Logic: Decrement in inventory
            setInventory(prev => {
                return prev.map(i => {
                    if (i.id === data.id) {
                        return { ...i, count: i.count - 1 };
                    }
                    return i;
                }).filter(i => i.count > 0);
            });

            // Clone and ensure distinct (reset count to 1 for board instance)
            itemToPlace = { ...inventoryItem, count: 1 };

        } else if (data.source === 'board') {
            itemToPlace = board[data.originIndex];
            if (!itemToPlace) return;

            // Remove from old position
            setBoard(prev => {
                const newBoard = { ...prev };
                delete newBoard[data.originIndex];
                return newBoard;
            });
        }

        // Place at new position
        if (itemToPlace) {
            setBoard(prev => ({ ...prev, [targetIndex]: itemToPlace }));
        }
    };

    const handleInventoryDrop = (e) => {
        e.preventDefault();
        const dataStr = e.dataTransfer.getData('application/json');
        if (!dataStr) return;

        const data = JSON.parse(dataStr);

        if (data.source === 'board') {
            const itemOnBoard = board[data.originIndex];
            if (!itemOnBoard) return;

            // Remove from board
            setBoard(prev => {
                const newBoard = { ...prev };
                delete newBoard[data.originIndex];
                return newBoard;
            });

            // Add back to inventory
            setInventory(prev => {
                const existingIndex = prev.findIndex(i => i.id === itemOnBoard.id);
                if (existingIndex >= 0) {
                    const newInv = [...prev];
                    newInv[existingIndex] = {
                        ...newInv[existingIndex],
                        count: newInv[existingIndex].count + 1
                    };
                    return newInv;
                } else {
                    return [...prev, { ...itemOnBoard, count: 1 }];
                }
            });
        }
    };

    // --- Form Handlers ---

    const handleGenerate = () => {
        setIsGenerating(true);
        setGeneratedOutput(null);

        // Simulate network request or processing delay
        setTimeout(() => {
            const placedItems = Object.entries(board).map(([indexStr, item]) => {
                const index = parseInt(indexStr);
                return {
                    id: item.id,
                    x: Math.floor(index / cols),
                    y: index % cols
                };
            });

            const output = {
                config: {
                    age: parseInt(formData.age),
                    language: formData.language,
                    narrator: formData.narrator
                },
                gridState: placedItems
            };

            setGeneratedOutput(JSON.stringify(output, null, 2));
            setIsGenerating(false);
        }, 1500); // 1.5 second simulated delay
    };

    // --- Render Helpers ---

    const cellStyle = {
        width: 'var(--cell-size, 80px)',
        height: 'var(--cell-size, 80px)',
    };

    return (
        <div className="flex flex-col md:flex-row gap-8 max-w-6xl mx-auto h-full p-4 font-sans text-slate-800">

            {/* Board Section */}
            <div className="flex flex-col items-center flex-[2] min-w-0">
                <div className="w-full p-6 bg-white rounded-xl shadow-sm border border-slate-200 flex flex-col items-center">
                    <div
                        className="grid bg-slate-400 gap-[2px] border-4 border-slate-400 rounded-lg shadow-md select-none"
                        style={{
                            gridTemplateColumns: `repeat(${cols}, var(--cell-size, 80px))`,
                            gridTemplateRows: `repeat(${rows}, var(--cell-size, 80px))`,
                        }}
                    >
                        {Array.from({ length: rows * cols }).map((_, i) => {
                            const item = board[i];
                            return (
                                <div
                                    key={i}
                                    className={`bg-white relative flex items-center justify-center transition-colors ${!item ? 'hover:bg-slate-50' : ''}`}
                                    style={cellStyle}
                                    onDragOver={handleDragOver}
                                    onDrop={(e) => handleBoardDrop(e, i)}
                                    data-index={i}
                                >
                                    {item && (
                                        <div
                                            draggable
                                            onDragStart={(e) => handleDragStart(e, item, 'board', i)}
                                            className="w-full h-full flex items-center justify-center text-4xl cursor-grab active:cursor-grabbing hover:scale-110 transition-transform"
                                        >
                                            {item.icon}
                                        </div>
                                    )}
                                </div>
                            );
                        })}
                    </div>
                </div>

                {/* Controls Form */}
                <div className="w-full mt-6 p-6 bg-white rounded-xl shadow-sm border border-slate-200">
                    <div className="flex flex-wrap items-end justify-center gap-6">

                        <div className="flex flex-col gap-1.5">
                            <label className="text-sm font-semibold text-slate-500 flex items-center gap-1">
                                <Hash size={14} /> Age
                            </label>
                            <select
                                value={formData.age}
                                onChange={(e) => setFormData({ ...formData, age: e.target.value })}
                                className="px-3 py-2 bg-white border border-slate-300 rounded-md text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
                                disabled={isGenerating}
                            >
                                {[1, 2, 3, 4, 5, 6].map(n => <option key={n} value={n}>{n}</option>)}
                            </select>
                        </div>

                        <div className="flex flex-col gap-1.5">
                            <label className="text-sm font-semibold text-slate-500 flex items-center gap-1">
                                <Globe size={14} /> Language
                            </label>
                            <select
                                value={formData.language}
                                onChange={(e) => setFormData({ ...formData, language: e.target.value })}
                                className="px-3 py-2 bg-white border border-slate-300 rounded-md text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
                                disabled={isGenerating}
                            >
                                <option value="de">de</option>
                                <option value="ru">ru</option>
                                <option value="en">en</option>
                            </select>
                        </div>

                        <div className="flex flex-col gap-1.5">
                            <label className="text-sm font-semibold text-slate-500 flex items-center gap-1">
                                <User size={14} /> Narrator
                            </label>
                            <select
                                value={formData.narrator}
                                onChange={(e) => setFormData({ ...formData, narrator: e.target.value })}
                                className="px-3 py-2 bg-white border border-slate-300 rounded-md text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
                                disabled={isGenerating}
                            >
                                <option value="Male">Male</option>
                                <option value="Female">Female</option>
                            </select>
                        </div>

                        <button
                            onClick={handleGenerate}
                            disabled={isGenerating}
                            className={`px-6 py-2 flex items-center gap-2 font-semibold rounded-md shadow-sm transition-colors text-sm ${
                                isGenerating 
                                    ? 'bg-blue-400 cursor-not-allowed text-white' 
                                    : 'bg-blue-600 hover:bg-blue-700 text-white'
                            }`}
                        >
                            {isGenerating && <Loader2 className="animate-spin w-4 h-4" />}
                            {isGenerating ? 'Generating...' : 'Generate'}
                        </button>
                    </div>
                </div>

                <div className="w-full mt-6 p-6 bg-white rounded-xl shadow-sm border border-slate-200">
                    {generatedOutput ? (
                         <div className="relative">
                            <h3 className="text-sm font-bold text-slate-700 mb-2">Generated Output</h3>
                            <pre className="bg-slate-50 p-4 rounded-lg text-xs font-mono text-slate-700 overflow-x-auto border border-slate-200 max-h-60 overflow-y-auto">
                                {generatedOutput}
                            </pre>
                        </div>
                    ) : (
                        <p className="text-gray-500 text-sm italic text-center">
                            {isGenerating ? "Processing grid data..." : "Output console results will appear here."}
                        </p>
                    )}
                </div>
            </div>

            {/* Sidebar Section */}
            <div className="flex flex-col flex-1 min-w-[300px] bg-white rounded-xl shadow-lg border border-slate-100 max-h-[800px]">
                <div className="p-4 border-b border-slate-100">
                    <h2 className="text-lg font-bold text-slate-800 flex items-center gap-2">
                        <Settings className="w-5 h-5" /> Inventory
                    </h2>
                    <p className="text-xs text-slate-500 mt-1">Drag items to board. Drag back to remove.</p>
                </div>

                <div
                    className="flex-1 overflow-y-auto p-3 flex flex-col gap-3"
                    onDragOver={(e) => { e.preventDefault(); e.currentTarget.classList.add('bg-blue-50'); }}
                    onDragLeave={(e) => { e.currentTarget.classList.remove('bg-blue-50'); }}
                    onDrop={(e) => { e.currentTarget.classList.remove('bg-blue-50'); handleInventoryDrop(e); }}
                >
                    {inventory.length === 0 ? (
                        <div className="text-center p-8 text-slate-400 italic">Inventory empty</div>
                    ) : (
                        inventory.map(item => (
                            <div
                                key={item.id}
                                draggable
                                onDragStart={(e) => handleDragStart(e, item, 'inventory')}
                                className="relative flex items-center p-3 bg-white border border-slate-200 rounded-lg shadow-sm hover:shadow-md hover:border-blue-400 transition-all cursor-grab active:cursor-grabbing select-none"
                            >
                                <div className="w-10 h-10 flex items-center justify-center bg-slate-50 rounded-md text-2xl mr-3">
                                    {item.icon}
                                </div>
                                <div>
                                    <div className="font-semibold text-sm text-slate-800">{item.title}</div>
                                    <div className="text-xs text-slate-500">{item.desc}</div>
                                </div>
                                <div className="absolute top-2 right-2 bg-blue-600 text-white text-[10px] font-bold px-2 py-0.5 rounded-full shadow-sm">
                                    x{item.count}
                                </div>
                            </div>
                        ))
                    )}
                </div>
            </div>

            {/* CSS Variables for Responsive Cell Sizing */}
            <style>{`
        :root {
          --cell-size: 80px;
        }
        @media (max-width: 768px) {
          :root {
            --cell-size: 50px;
          }
        }
      `}</style>
        </div>
    );
};


// Default Export for Preview
export default function App() {
    return (
        <div className="min-h-screen bg-[#f0f2f5] py-8">
            <GridWidget />
        </div>
    );
}
